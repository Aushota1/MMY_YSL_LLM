interface bus_if #(
    parameter WIDTH = 8
);
    logic clk;
    logic rst_n;
    logic [WIDTH-1:0] addr;
    logic [WIDTH-1:0] data;
    logic read;
    logic write;
    logic ready;
    
    modport master (
        output addr, data, read, write,
        input ready, clk, rst_n
    );
    
    modport slave (
        input addr, data, read, write, clk, rst_n,
        output ready
    );
endinterface

module interface_wrapper (
    bus_if.master bus
);
    logic [7:0] internal_reg;
    
    always_ff @(posedge bus.clk or negedge bus.rst_n) begin
        if (!bus.rst_n)
            internal_reg <= '0;
        else if (bus.write)
            internal_reg <= bus.data;
    end
    
    assign bus.ready = 1'b1;
endmodule
