module alu_007 #(parameter W = 32) (
    input  logic [W-1:0] a, b,
    input  logic [2:0]   op,
    output logic [W-1:0] y,
    output logic zero, carry
);
    logic [W:0] result;
    always_comb case (op)
        3'b000: result = a + b;
        3'b001: result = a - b;
        3'b010: result = a & b;
        3'b011: result = a | b;
        3'b100: result = a ^ b;
        default: result = a;
    endcase
    assign y = result[W-1:0];
    assign carry = result[W];
    assign zero = (y == 0);
endmodule
